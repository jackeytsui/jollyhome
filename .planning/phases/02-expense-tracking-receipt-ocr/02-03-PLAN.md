    ```typescript
    import { useState, useEffect, useCallback } from 'react';
    import { supabase } from '@/lib/supabase';
    import { useHouseholdStore } from '@/stores/household';
    import type { Settlement, PaymentPreferences } from '@/types/expenses';
    ```

    Returns: `{ settlements, paymentPrefs, loading, createSettlement, loadSettlements, loadPaymentPrefs, updatePaymentPrefs }`

    `loadSettlements(otherUserId?: string)`:
    - Query `settlements` table for household, optionally filter by from_user_id/to_user_id matching current user and otherUserId
    - Order by created_at DESC

    `createSettlement(toUserId: string, amountCents: number, paymentMethod?: string, note?: string)`:
    - Insert into `settlements` table: `{ household_id, from_user_id: currentUserId, to_user_id: toUserId, amount_cents: amountCents, payment_method: paymentMethod, note }`
    - On success, reload settlements
    - Return boolean success

    `loadPaymentPrefs(userId: string)`:
    - Query `payment_preferences` table WHERE user_id = userId
    - Store in paymentPrefs state

    `updatePaymentPrefs(prefs: Partial<PaymentPreferences>)`:
    - Upsert into `payment_preferences` with user_id = currentUserId
    - Use `supabase.from('payment_preferences').upsert({ user_id: currentUserId, ...prefs }, { onConflict: 'user_id' })`
  </action>
