```sql
-- Top-level DDL: sandbox_data table
CREATE TABLE IF NOT EXISTS public.sandbox_data (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  household_id UUID NOT NULL REFERENCES public.households(id) ON DELETE CASCADE,
  data_type TEXT NOT NULL, -- 'expense', 'chore', 'meal', 'event', 'member'
  data JSONB NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Top-level DDL: RLS
ALTER TABLE public.sandbox_data ENABLE ROW LEVEL SECURITY;

CREATE POLICY "sandbox_data_read" ON public.sandbox_data FOR SELECT
  USING (household_id IN (SELECT household_id FROM public.household_members WHERE user_id = auth.uid()));

CREATE POLICY "sandbox_data_delete" ON public.sandbox_data FOR DELETE
  USING (household_id IN (SELECT household_id FROM public.household_members WHERE user_id = auth.uid() AND role = 'admin'));

CREATE POLICY "sandbox_data_insert" ON public.sandbox_data FOR INSERT
  WITH CHECK (household_id IN (SELECT household_id FROM public.household_members WHERE user_id = auth.uid() AND role = 'admin'));
```

**clear_sandbox_data function** — removes sandbox household and all related data:
```sql
CREATE OR REPLACE FUNCTION public.clear_sandbox_data(p_user_id UUID)
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER SET search_path = public
AS $$
BEGIN
  -- Delete sandbox households (CASCADE deletes members, invites, sandbox_data)
  DELETE FROM households
  WHERE created_by = p_user_id AND is_sandbox = true;

  -- Clear active_household_id if it was the sandbox
  UPDATE profiles
  SET active_household_id = (
    SELECT hm.household_id FROM household_members hm
    JOIN households h ON h.id = hm.household_id
    WHERE hm.user_id = p_user_id AND hm.status = 'active' AND h.is_sandbox = false
    LIMIT 1
  )
  WHERE id = p_user_id;
END;
$$;
```

```typescript
import Purchases, { LOG_LEVEL } from 'react-native-purchases';
import { Platform } from 'react-native';
import { RC_APPLE_KEY, RC_GOOGLE_KEY } from '@/constants/config';

export async function initRevenueCat(userId: string) {
  if (__DEV__) {
    Purchases.setLogLevel(LOG_LEVEL.VERBOSE);
  }
  if (Platform.OS === 'ios') {
    Purchases.configure({ apiKey: RC_APPLE_KEY, appUserID: userId });
  } else if (Platform.OS === 'android') {
    Purchases.configure({ apiKey: RC_GOOGLE_KEY, appUserID: userId });
  }
  // Web: RevenueCat not supported — use Stripe (deferred to later)
}

export async function getEntitlements() {
  try {
    const { entitlements } = await Purchases.getCustomerInfo();
    return {
      isPlus: entitlements.active['plus'] !== undefined,
      isPro: entitlements.active['pro'] !== undefined,
      isTrial: entitlements.active['plus']?.periodType === 'TRIAL' || entitlements.active['pro']?.periodType === 'TRIAL',
      trialDaysRemaining: calculateTrialDays(entitlements.active['plus'] ?? entitlements.active['pro']),
    };
  } catch {
    // RevenueCat not configured or not available (web)
    return { isPlus: false, isPro: false, isTrial: false, trialDaysRemaining: 0 };
  }
}

export async function hasPlusEntitlement(): Promise<boolean> {
  const { isPlus, isPro } = await getEntitlements();
  return isPlus || isPro;
}

function calculateTrialDays(entitlement: any): number {
  if (!entitlement?.expirationDate) return 0;
  const diff = new Date(entitlement.expirationDate).getTime() - Date.now();
  return Math.max(0, Math.ceil(diff / (1000 * 60 * 60 * 24)));
}
```
