```typescript
interface Profile {
  id: string;
  display_name: string | null;
  avatar_url: string | null;
  avatar_type: 'photo' | 'illustrated' | 'initials';
  dietary_preferences: string[]; // e.g., ['vegetarian', 'gluten-free']
  dietary_notes: string | null;  // e.g., "I don't eat pork or anything too spicy"
  biometric_enabled: boolean;
  theme_override: 'light' | 'dark' | 'system';
  language_override: string | null;
}

export function useProfile() {
  return {
    profile,           // Profile | null (current user's profile)
    loadProfile,       // () => Promise<void>
    updateProfile,     // (updates: Partial<Profile>) => Promise<void>
    uploadAvatar,      // (uri: string) => Promise<string> — uploads to Supabase Storage, returns URL
    isLoading,
    error,
  };
}
```

**loadProfile**: `supabase.from('profiles').select('*').eq('id', user.id).single()`
Use TanStack React Query `useQuery` with key `['profile', user.id]`.

**updateProfile**: `supabase.from('profiles').update(updates).eq('id', user.id)`
Invalidate query cache on success.

**uploadAvatar**: Use `expo-image-picker` to get local URI, then:
- Generate filename: `avatars/${user.id}/${Date.now()}.jpg`
- Upload to Supabase Storage bucket `avatars`
- Get public URL and update profile's `avatar_url`

2. Create `src/hooks/useMembers.ts`:

```typescript
interface Member {
  id: string;           // household_members.id
  user_id: string;
  role: 'admin' | 'member';
  status: 'active' | 'pending';
  joined_at: string;
  profile: {
    display_name: string | null;
    avatar_url: string | null;
    dietary_preferences: string[];
  };
}

export function useMembers(householdId: string | null) {
  return {
    members,             // Member[]
    loadMembers,         // () => Promise<void>
    removeMember,        // (memberId: string) => Promise<void>
    updateMemberRole,    // (memberId: string, role: 'admin' | 'member') => Promise<void>
    leaveHousehold,      // () => Promise<void>
    approveMember,       // (memberId: string) => Promise<void>
    isLoading,
    error,
  };
}
```

**loadMembers**: Join query:
```typescript
supabase
  .from('household_members')
  .select('*, profile:profiles(display_name, avatar_url, dietary_preferences)')
  .eq('household_id', householdId)
  .order('joined_at', { ascending: true })
```
Use TanStack React Query with key `['members', householdId]`.

**removeMember**:
- Delete from `household_members` where `id = memberId`
- Can only be done by admin (check `userRole === 'admin'` from householdStore)
- Invalidate members query

**updateMemberRole**:
- Update `household_members` set `role = newRole` where `id = memberId`
- Admin only

**leaveHousehold**:
- Delete own membership from `household_members`
- If user was last admin, warn (should promote someone first)
- Set `profiles.active_household_id = null`
- Clear household store
- Navigate to create-household or household picker
- Track: `posthog.capture('household_left')`

**approveMember**: Update `status` from `'pending'` to `'active'`
